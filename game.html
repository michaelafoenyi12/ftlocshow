let gamePhase = "question"; let selectedQuestion = null; let moneyBox = null; const questions = { 1: { text: "What is 2 + 2?", options: ["3", "4", "5", "6", "7"], correct: 1 }, 2: { text: "Which planet is known as the Red Planet?", options: ["Earth", "Mars", "Venus", "Jupiter", "Saturn"], correct: 1 }, 3: { text: "What color is the sky?", options: ["Blue", "Green", "Red", "Yellow", "Purple"], correct: 0 }, 4: { text: "What is the capital of France?", options: ["Berlin", "Madrid", "Paris", "Rome", "Lisbon"], correct: 2 } }; let timeLeft; let timerInterval; function loadQuestion(num) { if (gamePhase !== "question") return; const q = questions[num]; if (!q) { alert("No question added yet!"); return; } selectedQuestion = num; document.getElementById("gameArea").style.display = "block"; document.getElementById("questionText").innerText = q.text; const answersDiv = document.getElementById("answers"); answersDiv.innerHTML = ""; const letters = ["A", "B", "C", "D", "E"]; q.options.forEach((opt, index) => { const btn = document.createElement("button"); btn.innerText = letters[index] + ". " + opt; btn.onclick = () => checkAnswer(index, q.correct); answersDiv.appendChild(btn); }); startTimer(); } function startTimer() { timeLeft = 60; document.getElementById("timer").innerText = timeLeft; clearInterval(timerInterval); timerInterval = setInterval(() => { timeLeft--; document.getElementById("timer").innerText = timeLeft; if (timeLeft <= 0) { clearInterval(timerInterval); alert("Time's up!"); gamePhase = "question"; document.getElementById("gameArea").style.display = "none"; } }, 1000); } function checkAnswer(selected, correct) { clearInterval(timerInterval); if (selected === correct) { alert("Correct! Now select a money box."); gamePhase = "box"; showMoneyBoxes(); } else { alert("Wrong! Game Over."); } } function showMoneyBoxes() { const answersDiv = document.getElementById("answers"); answersDiv.innerHTML = ""; const title = document.createElement("h2"); title.innerText = "Box Challenge"; answersDiv.appendChild(title); moneyBox = Math.floor(Math.random() * 5); for (let i = 0; i < 5; i++) { const btn = document.createElement("button"); btn.innerText = "FTLOC " + (i + 1); btn.onclick = function () { clearInterval(timerInterval); if (i === moneyBox) { alert("🎉 YOU WIN! 💰"); } else { alert("❌ You Lost!"); } gamePhase = "question"; document.getElementById("gameArea").style.display = "none"; }; answersDiv.appendChild(btn); } startBoxTimer(); } function startBoxTimer() { timeLeft = 30; document.getElementById("timer").innerText = timeLeft; clearInterval(timerInterval); timerInterval = setInterval(() => { timeLeft--; document.getElementById("timer").innerText = timeLeft; if (timeLeft <= 0) { clearInterval(timerInterval); alert("⏰ Time's Up! Game Over."); gamePhase = "question"; document.getElementById("gameArea").style.display = "none"; } }, 1000); }
